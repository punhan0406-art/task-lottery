    doc,
    getDoc,
    setDoc,
    collection,
    getDocs,
    deleteDoc

const app = initializeApp(firebaseConfig);

// 🚀 ƏSAS FUNKSİYA
window.getTask = async function () {
    const name = document.getElementById("name").value.trim();

    if (!name) {
        alert("Adını yaz!");
        return;
    }

    const userRef = doc(db, "assignments", name);
    const userSnap = await getDoc(userRef);

    // əgər artıq tapşırıq alıbsa
    if (userSnap.exists()) {
        document.getElementById("result").innerText =
            "Sənin tapşırığın: " + userSnap.data().task;
        return;
    }

    // tasks-ları götür
    const tasksSnapshot = await getDocs(collection(db, "tasks"));

    if (tasksSnapshot.empty) {
        document.getElementById("result").innerText =
            "Tapşırıq qalmayıb.";
        return;
    }

    const tasks = [];

    tasksSnapshot.forEach((d) => {
        tasks.push({
            id: d.id,
            task: d.data().task
        });
    });

    // random seç
    const randomTask = tasks[Math.floor(Math.random() * tasks.length)];

    // user-ə yaz
    await setDoc(userRef, {
        task: randomTask.task
    });

    // tasks-dan sil
    await deleteDoc(doc(db, "tasks", randomTask.id));

    document.getElementById("result").innerText =
        "Sənin tapşırığın: " + randomTask.task;
};
